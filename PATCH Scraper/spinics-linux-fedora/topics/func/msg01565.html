Attached version adds a config option called allow_unknown_minions (False by default) and renames extend_hostname_to_fqdn to get_all_host_aliases which returns a flat list instead of a tuple. Tested with 

import func.overlord.client as fc

for i in ["*", "www.planet-lab.org", "planetlab-02.cs.princeton.edu", "www", "planetlab-02", "www;planetlab-02", "www.planet-lab.org;planetlab-02.cs.princeton.edu"]:
     for j in [True, False]:
         try:
             print i,j
             my_overlord = fc.Overlord(i, delegate=j)
             print my_overlord.command.run("true")
         except:
             print "ERROR"
         print "\n\n"
